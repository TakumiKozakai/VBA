Sub Copy()    'ディレクトリのパスを取得    Const Path As String = "/Users/kozakaitakumi/Documents/01_仕掛中/VBA/"    'ディレクトリ内のブック名を取得    Dim buf As String    buf = Dir(Path & "*.xlsx")    'コピー先ブックのシート数取得（末尾にコピーするため枚数を把握）    Dim cnt As Long    cnt = Workbooks("【VBA】.xlsm").Sheets.Count    'ディレクトリ内のブック数分をシートコピーを行う    Do While buf <> ""        'コピー元ブックを開く        Workbooks.Open Path & buf        'コピー先ブックのシート末尾にコピー元ブックの"Eclipse"シートをコピーする        Workbooks(buf).Worksheets("Eclipse").Copy _            After:=Workbooks("【VBA】.xlsm").Worksheets(cnt)        'コピーしたシートのシート名をブック名に変更        Workbooks("【VBA】.xlsm").Worksheets("Eclipse").Select        Workbooks("【VBA】.xlsm").Worksheets("Eclipse").Name = buf        Range = Range.Find(What:="E")        MsgBox (Cells)        MsgBox (Cells.Row)        MsgBox (Cells.Column)        Cell = Cells(Cell.Row, Cell.Column + 1)        MsgBox (Cell)                'コピー元ブックを閉じる        Workbooks(buf).Close        '次のブック名を取得        buf = Dir()        'シート数カウントアップ        cnt = cnt + 1    LoopEnd Sub
